CREATE TABLE USER (

USER_ID INT(10) NOT NULL AUTO_INCREMENT,

NAME VARCHAR(30) NOT NULL,

PRIMARY KEY(USER_ID)

);

CREATE TABLE FOOD_DRINK_DATABASE (

NAME VARCHAR(30) NOT NULL,

SERVING_SIZE DOUBLE UNSIGNED NOT NULL,

PROTEIN_PER_SERVING DOUBLE UNSIGNED NOT NULL,

CARBS_PER_SERVING DOUBLE UNSIGNED NOT NULL,

FAT_PER_SERVING DOUBLE UNSIGNED NOT NULL,

CALORIES_PER_SERVING DOUBLE UNSIGNED NOT NULL,

PRIMARY KEY(NAME)
);

CREATE TABLE WORKOUT_DATABASE (

NAME VARCHAR(30) NOT NULL,

TYPE VARCHAR(20) NOT NULL,

CALORIES_BURNED DOUBLE UNSIGNED NOT NULL,

PRIMARY KEY(NAME)
);

CREATE TABLE WORKOUT (

WORKOUT_ID INT(10) NOT NULL AUTO_INCREMENT,

NAME VARCHAR(30) NOT NULL,

DURATION TIME NOT NULL,

PRIMARY KEY(WORKOUT_ID),

FOREIGN KEY(NAME) REFERENCES WORKOUT_DATABASE(NAME)
);

CREATE TABLE FOOD_AND_DRINK (

MEAL_ID INT(10) NOT NULL AUTO_INCREMENT,

NAME VARCHAR(30) NOT NULL,

QUANTITY DOUBLE UNSIGNED NOT NULL,

PRIMARY KEY(MEAL_ID),

FOREIGN KEY(NAME) REFERENCES FOOD_DRINK_DATABASE(NAME)
);

CREATE TABLE WEIGHT (

LOG_ID INT(10) NOT NULL AUTO_INCREMENT,

WEIGHT DECIMAL UNSIGNED NOT NULL,

PRIMARY KEY(LOG_ID)
);

CREATE TABLE MICRONUTRIENT (

MICRO_NAME VARCHAR(30) NOT NULL,

REC_DAILY_AMOUNT DECIMAL UNSIGNED NOT NULL,

PRIMARY KEY(MICRO_NAME)
);

CREATE TABLE UNIT_CONVERSION (

MEASUREMENT_TYPE VARCHAR(20) NOT NULL,

CONVERSION_RATIO VARCHAR(40) NOT NULL,

PRIMARY KEY(MEASUREMENT_TYPE)
);
