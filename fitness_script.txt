CREATE TABLE USER (

USER_ID INT(10) NOT NULL AUTO_INCREMENT,

NAME VARCHAR(30) NOT NULL,

PRIMARY KEY(USER_ID)

);

CREATE TABLE NUTRITIONINFO (

NAME VARCHAR(30) NOT NULL,

SERVING_SIZE DOUBLE UNSIGNED NOT NULL,

PROTEIN DOUBLE UNSIGNED NOT NULL,

CARBS DOUBLE UNSIGNED NOT NULL,

FAT DOUBLE UNSIGNED NOT NULL,

CALORIES DOUBLE UNSIGNED NOT NULL,

IRON DOUBLE UNSIGNED NOT NULL,

CALCIUM DOUBLE UNSIGNED NOT NULL,

VITAMIN_A DOUBLE UNSIGNED NOT NULL,

VITAMIN_C DOUBLE UNSIGNED NOT NULL,

PRIMARY KEY(NAME)
);

CREATE TABLE WORKOUTINFO (

NAME VARCHAR(30) NOT NULL,

TYPE VARCHAR(20) NOT NULL,

CALORIES_BURNED_PER_MINUTE DOUBLE UNSIGNED NOT NULL,

PRIMARY KEY(NAME)
);

CREATE TABLE WORKOUT (

WORKOUT_ID INT(10) NOT NULL AUTO_INCREMENT,

NAME VARCHAR(30) NOT NULL,

DURATION INT  NOT NULL, 

INTENSITY VARCHAR(20) NOT NULL,

DATE DATE NOT NULL,

USER_ID INT(10),

PRIMARY KEY(WORKOUT_ID),

FOREIGN KEY(NAME) REFERENCES WORKOUTINFO(NAME),

FOREIGN KEY(USER_ID) REFERENCES USER(USER_ID)
);

CREATE TABLE FOOD_AND_DRINK (

MEAL_ID INT(10) NOT NULL AUTO_INCREMENT,

NAME VARCHAR(30) NOT NULL,

QUANTITY DOUBLE UNSIGNED NOT NULL,

CALORIES DOUBLE UNSIGNED NOT NULL,

DATE DATE NOT NULL,

USER_ID INT(10),

PRIMARY KEY(MEAL_ID),

FOREIGN KEY(NAME) REFERENCES NUTRITIONINFO(NAME),

FOREIGN KEY(USER_ID) REFERENCES USER(USER_ID)
);

CREATE TABLE WEIGHT (

LOG_ID INT(10) NOT NULL AUTO_INCREMENT,

NUMERIC_WEIGHT DECIMAL UNSIGNED NOT NULL,

DATE DATE NOT NULL,

USER_ID INT(10),

PRIMARY KEY(LOG_ID),

FOREIGN KEY(USER_ID) REFERENCES USER(USER_ID)
);
